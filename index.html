    // Funciones relacionadas con la ventana

    bool WindowShouldClose(void);                               // Comprueba si la aplicación debe cerrarse (tecla ESC presionada o clic en el icono de cerrar)
    bool IsWindowReady(void);                                   // Comprueba si la ventana se ha inicializado correctamente
    bool IsWindowFullscreen(void);                              // Comprueba si la ventana está actualmente en pantalla completa
    bool IsWindowHidden(void);                                  // Comprueba si la ventana está oculta
    bool IsWindowMinimized(void);                               // Comprueba si la ventana está minimizada
    bool IsWindowMaximized(void);                               // Comprueba si la ventana está maximizada
    bool IsWindowFocused(void);                                 // Comprueba si la ventana tiene el foco
    bool IsWindowResized(void);                                 // Comprueba si la ventana ha sido redimensionada en el último frame
    bool IsWindowState(unsigned int flag);                      // Comprueba si un flag específico de la ventana está activado
    void SetWindowState(unsigned int flags);                    // Establece el estado de configuración de la ventana usando flags
    void ClearWindowState(unsigned int flags);                  // Limpia los flags de estado de configuración de la ventana
    void ToggleFullscreen(void);                                // Alterna el estado de la ventana: pantalla completa/ventana, redimensiona el monitor
    void ToggleBorderlessWindowed(void);                        // Alterna el estado de la ventana: ventana sin bordes
    void MaximizeWindow(void);                                  // Maximiza la ventana, si es redimensionable
    void MinimizeWindow(void);                                  // Minimiza la ventana, si es redimensionable
    void RestoreWindow(void);                                   // Restaura la ventana (ni minimizada ni maximizada)
    void SetWindowIcon(Image image);                            // Establece el icono de la ventana (imagen única, RGBA 32bit)
    void SetWindowIcons(Image *images, int count);              // Establece los iconos de la ventana (múltiples imágenes, RGBA 32bit)
    void SetWindowTitle(const char *title);                     // Establece el título de la ventana
    void SetWindowPosition(int x, int y);                       // Establece la posición de la ventana en pantalla
    void SetWindowMonitor(int monitor);                         // Establece el monitor para la ventana actual
    void SetWindowMinSize(int width, int height);               // Establece el tamaño mínimo de la ventana (para FLAG_WINDOW_RESIZABLE)
    void SetWindowMaxSize(int width, int height);               // Establece el tamaño máximo de la ventana (para FLAG_WINDOW_RESIZABLE)
    void SetWindowSize(int width, int height);                  // Establece las dimensiones de la ventana
    void SetWindowOpacity(float opacity);                       // Establece la opacidad de la ventana [0.0f..1.0f]
    void SetWindowFocused(void);                                // Enfoca la ventana
    void *GetWindowHandle(void);                                // Obtiene el manejador nativo de la ventana
    int GetScreenWidth(void);                                   // Obtiene el ancho actual de la pantalla
    int GetScreenHeight(void);                                  // Obtiene el alto actual de la pantalla
    int GetRenderWidth(void);                                   // Obtiene el ancho de renderizado actual (considera HiDPI)
    int GetRenderHeight(void);                                  // Obtiene el alto de renderizado actual (considera HiDPI)
    int GetMonitorCount(void);                                  // Obtiene el número de monitores conectados
    int GetCurrentMonitor(void);                                // Obtiene el monitor actual donde se encuentra la ventana
    Vector2 GetMonitorPosition(int monitor);                    // Obtiene la posición del monitor especificado
    int GetMonitorWidth(int monitor);                           // Obtiene el ancho del monitor especificado
    int GetMonitorHeight(int monitor);                          // Obtiene el alto del monitor especificado
    int GetMonitorPhysicalWidth(int monitor);                   // Obtiene el ancho físico del monitor en milímetros
    int GetMonitorPhysicalHeight(int monitor);                  // Obtiene el alto físico del monitor en milímetros
    int GetMonitorRefreshRate(int monitor);                     // Obtiene la tasa de refresco del monitor especificado
    Vector2 GetWindowPosition(void);                            // Obtiene la posición XY de la ventana en el monitor
    Vector2 GetWindowScaleDPI(void);                            // Obtiene el factor de escala DPI de la ventana
    const char *GetMonitorName(int monitor);                    // Obtiene el nombre legible por humanos del monitor especificado
    void SetClipboardText(const char *text);                    // Establece el contenido de texto del portapapeles
    const char *GetClipboardText(void);                         // Obtiene el contenido de texto del portapapeles
    Image GetClipboardImage(void);                              // Obtiene la imagen del portapapeles
    void EnableEventWaiting(void);                              // Habilita la espera de eventos en EndDrawing(), sin sondeo automático
    void DisableEventWaiting(void);                             // Deshabilita la espera de eventos en EndDrawing(), sondeo automático activo

    // Funciones relacionadas con el cursor
    void ShowCursor(void);                                      // Muestra el cursor
    void HideCursor(void);                                      // Oculta el cursor
    bool IsCursorHidden(void);                                  // Comprueba si el cursor no es visible
    void EnableCursor(void);                                    // Habilita el cursor (lo desbloquea)
    void DisableCursor(void);                                   // Deshabilita el cursor (lo bloquea)
    bool IsCursorOnScreen(void);                                // Comprueba si el cursor está en la pantalla

    // Funciones relacionadas con el dibujo (Drawing)
    void ClearBackground(Color color);                          // Establece el color de fondo (limpia el framebuffer)
    void BeginDrawing(void);                                    // Prepara el canvas (framebuffer) para empezar a dibujar
    void EndDrawing(void);                                      // Finaliza el dibujo y asume el intercambio de buffers (doble buffer)
    void BeginMode2D(Camera2D camera);                          // Inicia el modo 2D con una cámara personalizada
    void EndMode2D(void);                                       // Finaliza el modo 2D
    void BeginMode3D(Camera3D camera);                          // Inicia el modo 3D con una cámara personalizada
    void EndMode3D(void);                                       // Finaliza el modo 3D y vuelve al modo ortográfico 2D por defecto
    void BeginTextureMode(RenderTexture2D target);              // Empieza a dibujar en una textura de renderizado
    void EndTextureMode(void);                                  // Finaliza el dibujo en la textura de renderizado
    void BeginShaderMode(Shader shader);                        // Inicia el dibujo con un shader personalizado
    void EndShaderMode(void);                                   // Finaliza el dibujo con shader (vuelve al shader por defecto)
    void BeginBlendMode(int mode);                              // Inicia el modo de mezcla (alpha, aditivo, multiplicado, etc.)
    void EndBlendMode(void);                                    // Finaliza el modo de mezcla (vuelve al valor por defecto: alpha)
    void BeginScissorMode(int x, int y, int width, int height); // Inicia el modo tijera (define un área de recorte en pantalla)
    void EndScissorMode(void);                                  // Finaliza el modo tijera
    void BeginVrStereoMode(VrStereoConfig config);              // Inicia el renderizado estéreo (requiere simulador VR)
    void EndVrStereoMode(void);                                 // Finaliza el renderizado estéreo

    // Funciones de configuración estéreo VR
    VrStereoConfig LoadVrStereoConfig(VrDeviceInfo device);     // Carga la configuración estéreo VR para un dispositivo
    void UnloadVrStereoConfig(VrStereoConfig config);           // Descarga la configuración estéreo VR

    // Funciones de gestión de Shaders
    Shader LoadShader(const char *vsFileName, const char *fsFileName);   // Carga shader desde archivos
    Shader LoadShaderFromMemory(const char *vsCode, const char *fsCode); // Carga shader desde cadenas de texto
    bool IsShaderValid(Shader shader);                                   // Comprueba si un shader es válido (cargado en GPU)
    int GetShaderLocation(Shader shader, const char *uniformName);       // Obtiene la ubicación de un uniform del shader
    int GetShaderLocationAttrib(Shader shader, const char *attribName);  // Obtiene la ubicación de un atributo del shader
    void SetShaderValue(Shader shader, int locIndex, const void *value, int uniformType);               // Establece valor de uniform
    void SetShaderValueV(Shader shader, int locIndex, const void *value, int uniformType, int count);   // Establece vector de valores uniform
    void SetShaderValueMatrix(Shader shader, int locIndex, Matrix mat);         // Establece matriz 4x4 en uniform
    void SetShaderValueTexture(Shader shader, int locIndex, Texture2D texture); // Establece textura en uniform
    void UnloadShader(Shader shader);                                    // Descarga el shader de la memoria GPU (VRAM)

    // Funciones de espacio de pantalla / mundo
    Ray GetScreenToWorldRay(Vector2 position, Camera camera);         // Obtiene un rayo desde la posición de pantalla (ej. ratón)

    Vector2 GetWorldToScreen2D(Vector2 position, Camera2D camera);    // Posición de pantalla para cámara 2D
    Vector2 GetScreenToWorld2D(Vector2 position, Camera2D camera);    // Posición de mundo para cámara 2D
    Matrix GetCameraMatrix(Camera camera);                            // Obtiene la matriz de transformación de la cámara

    // Funciones de tiempo (Timing)

    void WaitTime(double seconds);                              // Detiene la ejecución durante un tiempo

    int GetRandomValue(int min, int max);                       // Valor aleatorio entre min y max (ambos incluidos)
    int *LoadRandomSequence(unsigned int count, int min, int max); // Carga secuencia aleatoria sin repeticiones
    void UnloadRandomSequence(int *sequence);                   // Descarga la secuencia aleatoria
